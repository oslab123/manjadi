# Set your OpenAI API key
OPENAI_API_KEY="your_api_key_here"

# Set the API endpoint
API_ENDPOINT="https://api.openai.com/v1/chat/completions"

# Set the model name
MODEL_NAME="gpt-3.5-turbo"

# Get the user message from the command line argument
USER_MESSAGE="$1"

# Check if the user message is provided
if [ -z "$USER_MESSAGE" ]; then
  echo "Usage: $0 <message>"
  exit 1
fi

# Set the temperature
TEMPERATURE=0.1

# Build the JSON data
JSON_DATA='{
  "model": "'$MODEL_NAME'",
  "messages": [{"role": "user", "content": "'$USER_MESSAGE'"}],
  "temperature": '$TEMPERATURE'
}'

# Make the curl request and capture the result
RESULT=$(curl -s -X POST $API_ENDPOINT \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer $OPENAI_API_KEY" \
  -d "$JSON_DATA")

# Extract and display only the content
CONTENT=$(echo "$RESULT" | awk -F'"content":' '{print $2}' | tr -d '}' | tr -d '"')
echo "Content:"
echo $CONTENT
